1 Overview

This R Markdown contains only the CHD1 KO NPC RNA-seq analyses used for the manuscript section on androgen rescue of CHD1-deficiency-associated transcriptional changes.

This script generates:

  • DESeq2 results for:
    • CHD1 KO vs control, DMSO-treated NPCs
    • DHT vs DMSO in CHD1 KO NPCs
    • DHT + enzalutamide vs DHT in CHD1 KO NPCs
    • DHT + enzalutamide vs DMSO in CHD1 KO NPCs
  • QC plots for the RNA-seq dataset
  • Gene lists for downstream interpretation
  • Log2FC correlation plots used for the DHT rescue and enzalutamide analyses
  • Correlation comparison statistics testing whether enzalutamide attenuates DHT rescue

samples <- dir(path = "kallisto_output/", pattern = "_Kallisto")

sample.sheet <- data.frame(samples = samples) %>%
  mutate(
    replicate = sub("^(A[0-9]+)_.*", "\\1", samples),
    genotype = sub("^A[0-9]+_([^_]+)_.*", "\\1", samples),
    treatment = sub("^A[0-9]+_[^_]+_([^_]+)_.*", "\\1", samples),
    treatment = ifelse(grepl("D_E", samples), "D_E", treatment),
    genotype = ifelse(grepl("CHD1", samples), "CHD1_KO", genotype),
    group = factor(paste(genotype, treatment, sep = "_")),
    replicate = factor(replicate)

add_gene_symbols <- function(res) {
  res.df <- as.data.frame(res) %>%
    rownames_to_column("ensembl_id") %>%
    mutate(
      gene = gene.annotation$SYMBOL[
        match(ensembl_id, gene.annotation$GENEID)
      ]
    ) %>%
    arrange(pvalue) %>%
    dplyr::filter(!is.na(padj))

  return(res.df)
}

8 Plotting functions

theme_chd1_paper <- function(base_size = 8) {
  theme_classic(base_size = base_size) +
    theme(
      text = element_text(family = "Arial", colour = "black"),
      axis.title = element_text(size = base_size, colour = "black"),
      axis.text = element_text(size = base_size - 1, colour = "black"),
      axis.line = element_line(linewidth = 0.4, colour = "black"),
      axis.ticks = element_line(linewidth = 0.4, colour = "black"),
      plot.title = element_text(size = base_size + 1, hjust = 0.5),
      panel.border = element_blank(),
      legend.position = "none"
    )
}

plot_lfc_correlation <- function(res.x,
                                 res.y,
                                 xlab,
                                 ylab,
                                 title,
                                 file,
                                 decile = 1,
                                 lims = NULL) {

  x.decile <- res.x %>% dplyr::filter(pval.decile == decile)
  y.decile <- res.y %>% dplyr::filter(pval.decile == decile)

  lfc.compare <- inner_join(
    x.decile,
    y.decile,
    by = "ensembl_id",
    suffix = c("_x", "_y")
  ) %>%
    dplyr::filter(!is.na(log2FoldChange_x), !is.na(log2FoldChange_y))

  cor.stats <- cor.test(
    lfc.compare$log2FoldChange_x,
    lfc.compare$log2FoldChange_y,
    method = "pearson"
  )

  r.value <- unname(cor.stats$estimate)
  p.value <- cor.stats$p.value

  if (is.null(lims)) {
    max.lim <- ceiling(max(abs(c(
      lfc.compare$log2FoldChange_x,
      lfc.compare$log2FoldChange_y
    )), na.rm = TRUE))
    lims <- c(-max.lim, max.lim)
  }

  p.label <- ifelse(
    p.value < 2.2e-16,
    "'< 2.2e-16'",
    paste0("'= ", signif(p.value, 2), "'")
  )

  p <- ggplot(
    lfc.compare,
    aes(x = log2FoldChange_x, y = log2FoldChange_y)
  ) +
    geom_point(
      shape = 21,
      size = 1.4,
      stroke = 0.25,
      colour = "black",
      fill = "black"
    ) +
    geom_vline(xintercept = 0, linewidth = 0.35) +
    geom_hline(yintercept = 0, linewidth = 0.35) +
    geom_smooth(
      method = lm,
      se = TRUE,
      colour = "#E69F00",
      fill = "grey75",
      linewidth = 0.7
    ) +
    coord_fixed(xlim = lims, ylim = lims) +
    theme_chd1_paper(base_size = 8) +
    ggtitle(paste0(title, " (n = ", nrow(lfc.compare), ")")) +
    xlab(xlab) +
    ylab(ylab) +
    annotate(
      "text",
      x = Inf,
      y = Inf,
      hjust = 1.1,
      vjust = 1.4,
      size = 3,
      label = paste0(
        "italic(r) == ", round(r.value, 3),
        "*','~~italic(p)~", p.label
      ),
      parse = TRUE
    )

  print(p)

  ggsave(
    file,
    p,
    width = 3.2,
    height = 2.6,
    units = "in",
    device = cairo_pdf
  )

  return(list(
    data = lfc.compare,
    plot = p,
    r = r.value,
    p = p.value,
    n = nrow(lfc.compare)
  ))
}

all.lfc <- list(
  genotype = res.genotype.df %>%
    dplyr::select(ensembl_id, genotype_lfc = log2FoldChange, pval.decile),
  dht = res.dht.df %>%
